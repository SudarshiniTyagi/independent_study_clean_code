def get_line_and_plot(X, Y):
    z = np.polyfit(X, Y, 1)
    p = np.poly1d(z)
    xp = np.linspace(0, 22, 100)
    plt.plot(X,Y,'.',xp, p(xp),'-')
    plt.show()
    return z

def plot_all(data):
    uniqueDiets = chickweight.Diet.unique()
    for diet in uniqueDiets:

        df_this_diet_values = df_this_diet.values
        X = np.array(df_this_diet_values[:,1], dtype=float)
        Y = np.array(df_this_diet_values[:,0], dtype=float)
        z = get_line_and_plot(X, Y)
        print("Regression Line for diet ", diet, ": weight = (",round(z[0], 2),"* time) + ", round(z[1],2))
